    Примечание: по данному коду сортировка осуществляется от меньшего к большему

    Код

                    Random r = new Random();
                int n = 10;
                int v = n;
                int[] array = new int[n];
                for (int i = 0; i < n; i++)

                    array[i] = r.Next(-10, 11);
                    Console.Write($"[{ i}]={ array[i]}; ");
                }
                Console.WriteLine();
                Console.WriteLine($"Сортировка пузырьком");
                //Переменные для сортировки
                int a = 0;
                int b = 0;
                //переменная отвечающая за этап сортировки
                int e = 1;
                //провереть весь массив от сортировался
                while(n != 1)
                {
                    int count = 0;
                    Console.WriteLine();
                    Console.WriteLine($"Этап {e} сортировки");
                    //меняем местами меньший влево больший вправо
                    for(int i =0; i array[j])
                            {
                                a = array[i];
                                b = array[j];
                                array[i] = b;
                                array[j] = a;
                            }
                        }
                    }
                    //выводим по этапно отсортированный массив
                    for (int i = 0; i < n; i++)
                    {
                        Console.Write($"[{i}]={ array[i]}; ");
                    }
                    //увеличиваем переменную этап
                    e++;
                    //проверка последнего элемента на самое большое значение из оставшихся элементов
                    for (int i = 0; i < n; i++)
                    {
                        //последний элемент не должен проверяться, проверяются только перед последним элементом
                        if(i != n - 1)
                        {
                            //проверка последовательно на, то больше или равен ли последний элемент, текущего в цикле
                            if (array[n - 1] >= array[i])
                            {
                                count++;
                            }
                        }
                    }
                    //если последний элемент больше по значению, чем все другие элементы, то последний элемент отсекатся и последним элементом становится предыдущий элемент
                    if (count == n - 1)
                    {
                        n--;
                    }
                }
                //вывод отсортированного массива
                Console.WriteLine();
                for (int i = 0; i < v; i++)
                {
                    Console.Write($"[{i}]={ array[i]}; ");
