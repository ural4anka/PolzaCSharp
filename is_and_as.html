Оператор AS:

- оператор преобразует объект с одним классом в другой класс

- оператор преобразует переменную однового типа данных в другой

    using System;
    using System.Collections.Generic;

    namespace ConsoleApp3
    {
        class Program
        {
            static void Main(string[] args)
            {
                List robots = new List();
                robots.Add(new Killer("Bob", 800, new byte[] { 0, 0, 10}, 100));
                robots.Add(new Killer("Joe", 600, new byte[] { 0, 0, 10}, 100));
                robots.Add(new Killer("Baty", 800, new byte[] { 0, 0, 10}, 100));
                robots.Add(new Killer("Mary", 600, new byte[] { 0, 0, 10}, 100));
                
                Robot newRobot = null;

                foreach(Killer obj in robots)
                {
                    if(obj.Name == "Joe"){
                        newRobot = obj as Robot;
                    }
                    Console.WriteLine(obj.Name);
                }
            }
        }
    }
