Обработчик исключений | Оператор try-catch

Объявление оператора try-catch:

    try
    {
        код который проверяется
    }
    catch (ошибка при проверки кода)
    {
        действие, если нашлась эта ошибка
    }
    catch (ошибка при проверки кода)
    {
        действие, если нашлась эта ошибка
    }
    finally
    {
        Последнее действие, которое будет всегда выполняться
        Например: выход из базы данных
    }

Пример:

    try
    {
        int a, b, res;
        Console.Write("Введите a:");
        a = Convert.ToInt32(Console.ReadLine());
        Console.Write("Введите b:");
        b = Convert.ToInt32(Console.ReadLine());
        res = a / b;
        Console.WriteLine("Результат"+res);
    }
    catch (DivideByZeroException)
    {
        Console.WriteLine("Деление на ноль");
    }
    catch (FormatException)
    {
        Console.WriteLine("Вы вели не тот формат");
    }
    finally
    {
        Console.WriteLine("Выход из базы данных");
    }
